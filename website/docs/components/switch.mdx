import { Switch, Card, Table } from 'doom-design-system';
import { useState } from 'react';

# Switch

A control that allows the user to toggle between two states.

## Live Demo

export const SwitchDemo = () => {
  const [active, setActive] = useState(false);
  return (
    <Card style={{ padding: '2rem', display: 'flex', alignItems: 'center', gap: '1rem' }}>
      <Switch 
        checked={active} 
        onChange={setActive} 
        label={active ? "Active Mode" : "Inactive Mode"} 
      />
    </Card>
  );
}

<SwitchDemo />

## Usage

```tsx
import { Switch } from 'doom-design-system';
import { useState } from 'react';

export default function MyComponent() {
  const [enabled, setEnabled] = useState(false);

  return (
    <Switch 
      checked={enabled} 
      onChange={setEnabled} 
      label="Toggle me" 
    />
  );
}
```
